# [AGENT] POLLINATIONS CLOUD (Free, Smart, Detailed)
# FEATURES: Pointwise History + Cinematic Visual Prompts
import json
import re
import requests
import urllib.parse
import wikipediaapi
from core.narrative import NarrativePlanner
from core.cultural_intelligence_builder import CulturalIntelligenceBuilder
from core.data_service import GlobalCultureManager

class StoryAgent:
    def __init__(self, api_keys: list):
        self.planner = NarrativePlanner(api_key="none")
        self.cultural_builder = CulturalIntelligenceBuilder()
        self.culture_manager = GlobalCultureManager()
        self.wiki = wikipediaapi.Wikipedia(user_agent='CulturalApp/Cloudv2', language='en')

    def get_real_facts(self, country):
        try:
            page = self.wiki.page(f"History of {country}")
            if not page.exists(): page = self.wiki.page(country)
            if page.exists(): return page.summary[:1000].replace('\n', ' ')
        except: pass
        return f"History of {country}"

    def format_history_points(self, text):
        """Converts a paragraph into clean bullet points."""
        # Split by periods to get sentences
        sentences = [s.strip() for s in text.split('.') if len(s) > 10]
        # Take the top 3 clearest sentences
        selected = sentences[:3]
        # Join with bullet points
        return "\n• " + "\n• ".join(selected) + "."

    def ask_pollinations_cloud(self, prompt):
        """Hits the Free Pollinations Text API."""
        try:
            safe_prompt = urllib.parse.quote(prompt)
            url = f"https://text.pollinations.ai/{safe_prompt}?model=openai"
            response = requests.get(url, timeout=10)
            return response.text
        except Exception as e:
            print(f"[CLOUD ERROR] {e}")
            return "{}"

    def clean_json(self, text):
        try:
            match = re.search(r'\{.*\}', text, re.DOTALL)
            if match: return json.loads(match.group(0))
        except: pass
        return {}

    def process(self, lat, lon, text="", era="Modern"): 
        # 1. Get Location & Facts
        profile = self.culture_manager.get_profile_by_coords(lat, lon)
        country = profile.get("Country", "Unknown")
        raw_facts = self.get_real_facts(country)
        
        # Format facts into bullet points for the dashboard
        formatted_history = self.format_history_points(raw_facts)

        # 2. Ask the Cloud (Dynamic Era)
        prompt = f"""
        Act as a JSON API. 
        Analyze: {country} during the year {era}.
        Theme: {text}
        Context: {raw_facts[:300]}
        
        Task:
        1. Write a 2-sentence cultural story about {country} set in {era}.
        2. Write a highly detailed, photorealistic image prompt. Include specific clothing, architecture, and lighting of {era}.
        
        Return JSON ONLY:
        {{
            "story": "...",
            "image_prompt": "..."
        }}
        """

        print(f"[CLOUD] Generating content for {country} ({era})...")
        raw_text = self.ask_pollinations_cloud(prompt)
        res_data = self.clean_json(raw_text)

        # Fallback
        if not res_data.get("story"):
            res_data = {
                "story": f"In {era}, {country} stood as a testament to resilience. {raw_facts[:100]}...",
                "image_prompt": f"Cinematic shot of {country} in {era}, historical architecture, 8k resolution, dramatic lighting"
            }

        # 3. Build Result
        intel = self.cultural_builder.build(lat, lon, country, profile, {})
        intel_dict = intel.model_dump() if hasattr(intel, 'model_dump') else intel.dict()
        
        # Pass the "Rich" prompt to the frontend
        intel_dict['video_prompt'] = res_data.get('image_prompt', f"View of {country}")
        intel_dict['image_url'] = res_data.get('image_prompt', f"View of {country}")
        
        # Inject the Bullet Points here
        intel_dict['cultural_history'] = formatted_history
        intel_dict['thought_signature'] = f"Pollinations Cloud ({era})"
        
        return res_data.get('story', 'No story generated.'), intel_dict
